## Usage
Show `success` toast

```jsx
import React, { useCallback } from 'react'
import { Button, toastService } from '@city/ui-kit'

export const Component = () => {
  const onClick = useCallback(() => {
    toastService.success('Success')
  }, [])

  return <Button onClick={onClick} variant="primary">Show success toast</Button>
}
```

import React, { useCallback } from 'react'
